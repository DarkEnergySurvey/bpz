This file describes how to run BPZ on a small example file in the test directory

----- How to format data for BPZ -----

format data files for des like:

%>python
f = open(file_path + '.cat', 'w')

f.write('# COADD_OBJECTS_ID\tMAG_AUTO_G\tMAG_AUTO_R\tMAG_AUTO_I\tMAG_AUTO_Z\tMAGERR_AUTO_G\tMAGERR_AUTO_R\tMAGERR_AUTO_I\tMAGERR_AUTO_Z\Z\n')

-- Z is true redshift
kys1 = ['COADD_OBJECTS_ID','MAG_AUTO_G','MAG_AUTO_R','MAG_AUTO_I','MAG_AUTO_Z',
       'MAGERR_AUTO_G','MAGERR_AUTO_R','MAGERR_AUTO_I','MAGERR_AUTO_Z','REDSHIFT']
for j in range(len(rs_data)):
    dt = rs_data[j]
    f.write('\t'.join(['{:}'.format(dt[k]) for k in kys1]) + '\n')
f.close()
